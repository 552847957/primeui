animate boolean null When enabled, displays an animation on click of the slider bar.
orientation string horizontal Orientation of the slider, valid values are horizontal and vertical.
range boolean false When speficed, allows two boundary values to be picked.

$(function () {
    $('#default').val(0);
    
    $('#value').text(50);
    
    $('#basic').puislider({
        onChange: function(event,obj) {
            $('#default').val(obj.value);
        }
    });
    
    $('#input').puislider({
        value: 50,
        onChange: function(event,obj) {
            $('#value').text(obj.value);
        }
    });
    
    $('#animate').puislider({
        animate: true
    });
    
    $('#step').puislider({
        step: 10
    });
    
    $('#range').puislider({
        range: true,
        value: [20,80]
    });
    
    $('#vertical').puislider({
        orientation: 'vertical'
    });
    
    $('#default').puiinputtext(); 

    $('#doc').puitabview();
});
