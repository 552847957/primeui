var DataDemo = {
    
    loadLazyCars: function (callback, ui) {
        var uri = 'rest/cars/lazylist/' + ui.first;

        $.ajax({
            type: "GET",
            url: uri,
            dataType: "json",
            context: this,
            success: function (response) {
                callback.call(this, response);
            }
        });
    }
    
};
