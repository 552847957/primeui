$('#trmnl').puiterminal({
    welcomeMessage: 'Welcome to PrimeUI Terminal, how are you today?',
    handler: function(request, response) {
        $.ajax({
            type: "GET",
            url: 'rest/terminal/' + request,
            dataType: "text",
            context: this,
            success: function(data) {
                response.call(this, data);
            }
        });
    }
});
